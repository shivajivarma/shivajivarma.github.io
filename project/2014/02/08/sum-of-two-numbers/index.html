Application to calculate sum of two number

Posted on February 8, 2014 by Shivaji Varma

Short and sweet application. Fetches values of a & b using GET Method [from URL] and computes there sum.

Output (A+B):

Code

$(document).ready(function(){
	var a = getUrlParameter("a");
	var b = getUrlParameter("b");
	if(a!="" && b!=""){
		if($.isNumeric(a) && $.isNumeric(b)){
			$("#output").html(+a + +b);
		}
		else  $("#output").html("Enter only numbers.");
	}
	else $("#output").html("Please fill the input and hit add button.");
});
function getUrlParameter( key ){  
		key = key.replace(/[\[]/,"\\\[").replace(/[\]]/,"\\\]");  
		var regexExpression = "[\\?&]"+key+"=([^&#]*)";  
		var regex = new RegExp( regexExpression );  
		var results = regex.exec( window.location.href );  
		if( results == null )    return "";  
			else    return unescape(results[1]).replace(/.*<\/script>/g, "");
}
